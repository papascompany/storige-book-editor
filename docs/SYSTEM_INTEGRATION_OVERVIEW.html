v2.6 v2.6 (2026-06-07) — 스프레드 책 인쇄 무결성

PRINT INTEGRITY SPREAD SNAPSHOT BOOKMOA 영향

스프레드(펼침 표지) 책의 편집완료 → 합성 → 출력 전 구간을 단일 소스로 묶고, 템플릿 사양부터 실제 cover.pdf 까지의 폭을 검증하는 무결성 체인을 도입했습니다.

🔥
편집완료 '프리즈' 근본수정
렌더러 프리즈는 오진이었습니다. 실제 원인은 (1) 커버 업로드 metadata 400(@Transform) + (2) 내지 unit TypeError(옵셔널 체이닝) 2건으로, 두 결함을 모두 수정했습니다.
📸
출력재현 단일소스 — EditSession 스냅샷
편집기 완료 시 EditSession.metadata.spread / metadata.spine 스냅샷을 저장합니다. 완료 검증 게이트를 mode === SPREAD → metadata.spread 존재 여부로 전환했습니다.
📄
스프레드 책 = 분리 2파일
API compose-mixed 가 스프레드 책에 outputMode='separate' 를 강제합니다 → cover.pdf(펼침면 전체: 뒷표지 | 책등 | 앞표지 + 날개) + content.pdf. 워커는 cover.pdf MediaBox ↔ 펼침면 총폭(metadata.spread) 무결성을 검증합니다(허용오차 max(0.2mm, 1px@dpi)).

무결성 체인

템플릿 권위(spreadConfig.spec) ⟵(B49) metadata.spread ⟵(P0-3) 실제 cover.pdf MediaBox. 이 체인으로 책등 가변폭 결함(cover 0.6mm 과대) 1건을 검출·수정했습니다.

  • 제본 페이지 가드(A13) — 무선 32p 최소 · 중철 64p 최대를 강제합니다(미설정 상품은 무영향).
  • woff2ToTtf 엔드포인트 신설 — PDF 텍스트 아웃라인화용 폰트 변환 경로 추가.
  • 운영 플래그 SPREAD_SNAPSHOT_HARD_FAIL — 기본 SOFT, true 설정 시 HARD 차단.
⚠️
Bookmoa 영향 (Breaking)
스프레드 책은 서버가 separate 출력을 강제합니다(single 로 전송해도 무시) → 합성 결과가 cover.pdf + content.pdf 2파일로 반환됩니다.

PHP API 호출 코드 예시

PHP · storige_common.php
<?php
// ① JWT 토큰 발급 (PHP 서버에서 호출)
function getShopSessionToken(int $orderSeqno, int $memberSeqno, string $templateSetId): string {
    $ch = curl_init(STORIGE_API_URL . '/auth/shop-session');
    curl_setopt_array($ch, [
        CURLOPT_POST            => true,
        CURLOPT_RETURNTRANSFER  => true,
        CURLOPT_HTTPHEADER      => ['Content-Type: application/json', 'X-API-Key: ' . STORIGE_API_KEY],
        CURLOPT_POSTFIELDS      => json_encode([
            'orderSeqno'    => $orderSeqno,
            'memberSeqno'   => $memberSeqno,
            'templateSetId' => $templateSetId,
        ]),
    ]);
    $res = json_decode(curl_exec($ch), true);
    curl_close($ch);
    return $res['token'] ?? '';
}

// ② PDF 파일 업로드
function uploadPdfFile(string $filePath, string $fileType, int $orderSeqno): array {
    $cFile = new CURLFile($filePath, 'application/pdf', basename($filePath));
    $ch = curl_init(STORIGE_API_URL . '/files/upload/external');
    curl_setopt_array($ch, [
        CURLOPT_POST           => true,
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_HTTPHEADER     => ['X-API-Key: ' . STORIGE_API_KEY],
        CURLOPT_POSTFIELDS     => ['file' => $cFile, 'type' => $fileType, 'orderSeqno' => $orderSeqno],
    ]);
    $res = json_decode(curl_exec($ch), true);
    curl_close($ch);
    return $res; // { id, fileUrl, fileSize }
}

// ③ PDF 검증 요청
function requestValidation(string $fileId, string $fileType, array $orderOptions, string $callbackUrl = ''): array {
    $body = ['fileId' => $fileId, 'fileType' => $fileType, 'orderOptions' => $orderOptions];
    if ($callbackUrl) $body['callbackUrl'] = $callbackUrl;
    $ch = curl_init(STORIGE_API_URL . '/worker-jobs/validate/external');
    curl_setopt_array($ch, [
        CURLOPT_POST           => true,
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_HTTPHEADER     => ['Content-Type: application/json', 'X-API-Key: ' . STORIGE_API_KEY],
        CURLOPT_POSTFIELDS     => json_encode($body),
    ]);
    $res = json_decode(curl_exec($ch), true);
    curl_close($ch);
    return $res; // { id: jobId, status: 'PENDING' }
}

// ④ Webhook 수신 처리 (validation_callback.php)
function handleValidationWebhook(): void {
    $payload = json_decode(file_get_contents('php://input'), true);
    switch ($payload['event'] ?? '') {
        case 'validation.completed':  updateOrderValidationStatus($payload['jobId'], 'PASS'); break;
        case 'validation.fixable':    updateOrderValidationStatus($payload['jobId'], 'FIXABLE', $payload['result']); break;
        case 'validation.failed':     notifyCustomerReupload($payload['jobId'], $payload['errorMessage']); break;
    }
    http_response_code(200);
    echo json_encode(['received' => true]);
}
